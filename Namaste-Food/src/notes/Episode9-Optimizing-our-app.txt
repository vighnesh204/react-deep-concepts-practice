🚀 Custom Hooks

Creating a custom hook is not a mandatory thing
but its a very good thing because that will make your code more readable, modular, reusable
that is the use of creating custom hooks

In our app 
    RestaurantMenu component
=> It has two major responsibilities 
    fetching the data 
    displaying the data on to the UI

=> It should only be concerned about display the data
=> It should not worry that where the data is coming from how the data is coming from

🚀 Before Custom hook 
    RestaurantMenu.jsx

import { useState } from "react";
import { useEffect } from "react";
import Shimmer from "./Shimmer";
import { useParams } from "react-router-dom";
import { MENU_API } from "../utils/constants";

const RestaurantMenu = () => {
  const [resInfo, setResInfo] = useState(null);
  const { resId } = useParams();

  useEffect(() => {
    fetchMenu();
  }, []);

  const fetchMenu = async () => {
    const data = await fetch(MENU_API + resId);

    const json = await data.json();
    console.log(json);
    setResInfo(json.data);
  };

  const { name, cuisines, costForTwo } = resInfo?.cards[0].card?.card?.info;
  //   resInfo?.cards[2]?.card?.card?.info;

  const { itemCards } =
    resInfo?.cards[2]?.groupedCard?.cardGroupMap?.REGULAR?.cards[1]?.card?.card;

  return resInfo === null ? (
    <Shimmer />
  ) : (
    <div className="max-w-xl mx-auto mt-10 p-6 bg-white shadow-sm rounded-lg">
      <h2 className="text-xl font-semibold text-gray-800 mb-1">{name}</h2>
      <h2 className="text-xl font-semibold text-gray-800 mb-1">
        {cuisines.join(", ")}
      </h2>
      <h2 className="text-xl font-semibold text-gray-800 mb-1">{costForTwo}</h2>

      <h4 className="text-gray-500 mb-6">Menu</h4>

      <ul className="space-y-3">
        {itemCards.map((item) => (
          <li className="p-3 bg-gray-50 rounded-md">
            {itemCards[0].card.info.name} - {" Rs."}
            {itemCards[0].card.info.price / 100 ||
              itemCards[0].card.info.defaultPrice / 100}
          </li>
        ))}
        {/* <li className="p-3 bg-gray-50 rounded-md">{itemCards[0].card.info.name}</li> */}
        <li className="p-3 bg-gray-50 rounded-md">Diet Coke</li>
        <li className="p-3 bg-gray-50 rounded-md">Mutton</li>
      </ul>
    </div>
  );
};

export default RestaurantMenu;
